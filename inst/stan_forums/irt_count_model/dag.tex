\begin{tikzpicture}[
	> = stealth, % arrow head style
	shorten > = 1pt, % don't touch arrow head to node
	auto,
	node distance = 2cm, % distance between nodes
	semithick, % line style
	U/.style={circle, draw=black, inner sep=1.8pt, outer sep=1.5pt, minimum size=3mm, font=\Large},  %draw=black, fill=white
	O/.style={circle, inner sep=3.4pt, outer sep=-.75pt, minimum size=3mm, font=\Large},    
	background rectangle/.style={fill=white}, show background rectangle  % add backgroud          %draw=white, fill=white
	]
	
	% Nodes and their relative positions
	% t = 0
    \node[O] (R) {$R$};
    \node[U] (A) [below left = .8 of R] {$A$};	
    \node[U] (D) [below right = .8 of R] {$D$};
    \node[O] (T) [below = 1.5 of R] {$T$};
       
	% Paths connecting nodes
	\path[->] (D) edge (R);
	\path[->] (A) edge (R);
	\path[->] (T) edge (D);
	\path[->] (T) edge[densely dashed] (R);
\end{tikzpicture}
